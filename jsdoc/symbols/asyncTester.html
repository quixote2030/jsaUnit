Class Detail
asyncTester()
The asyncTester module contains functionality to assist in UNIT testing asynchronous JavaScript 
Sample:

//execute async call
asyncTester.run(
    function () {
        jQuery.ajax({
            url: "/someApp/TestEndpoint",
            complete: function () {
                asyncTester.callback("myKey", arguments);
            }
        });
    })
    .whenDone(function () {
        //code to kick off tests/test runner
    });

//unit test	
function testShouldDoSomething () {
    asyncTester.test("myKey", function(xmlHttp, textStatus) {
        //run test assertions
    });
}
